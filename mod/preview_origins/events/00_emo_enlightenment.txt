namespace = emo
event = {
	id = emo.7
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		# "Endeavour to Enlightenment" Origin
		# on_leader_level_up (fires for level 1 too)
		if = {
			limit = {
				has_origin = origin_enlightenment
				from = {
					has_skill > 1
				}
			}
			# Add additional traits a little later
			from = {
				leader_event = {
					id = emo.14
				}
			}
		}
	}
}
