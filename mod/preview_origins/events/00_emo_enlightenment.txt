namespace = emo
event = {
	id = emo.7
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		# "Endeavour to Enlightenment" Origin
		# on_leader_level_up
		if = {
			limit = {
				has_origin = origin_enlightenment
			}
			if = {
				limit = {
					from = {
						NOT = {
							has_leader_flag = has_gained_level_trait
						}
					}
				}
				from = {
					# if the leader wasn't recruited normally, vanilla behaviour needs to be prevented here
					set_leader_flag = has_gained_level_trait
				}
			}
			from = {
				# level up logic
				if = {
					limit = {
						leader_class = ruler
						owner = {
							NOT = {
								has_ethic = "ethic_gestalt_consciousness"
							}
						}
					}
					random_list = {
						60 = {
						}
						40 = {
							modifier = {
								factor = 1.5
								has_level = 2
							}
							modifier = {
								factor = 2.5
								has_level = 3
							}
							modifier = {
								factor = 5
								has_level > 3
							}
							add_random_beneficial_leader_trait = yes
						}
					}
				}
				else_if = {
					limit = {
						NOT = {
							leader_class = ruler
						}
					}
					random_list = {
						80 = {
						}
						20 = {
							modifier = {
								factor = 2
								OR = {
									leader_class = governor
									leader_class = general
								}
							}
							modifier = {
								factor = 1.5
								has_level = 2
							}
							modifier = {
								factor = 2.5
								has_level = 3
							}
							modifier = {
								factor = 5
								has_level > 3
							}
							add_random_beneficial_leader_trait = yes
						}
					}
				}
			}
		}
	}
}

event = {

	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	immediate = {
		# "Endeavour to Enlightenment" Origin
		# on_game_start
		if = {
			limit = {
				has_origin = origin_enlightenment
			}
			# prevent vanilla level up behaviour
			every_pool_leader = {
				if = {
					limit = {

				}
			}
			every_owned_leader = {

						}
					}
					set_leader_flag = has_gained_level_trait

				}
			}
		}
	}
